\newglossaryentry{bytecode}{
  name=bytecode,
  description={Código independente da máquina que xeran compiladores de determinadas linguaxes (Java, Erlang,\dots) e que é executado polo correspondente intérprete}
}

\newglossaryentry{verifactu}{
  name=VeriFactu,
  description={Mandatory electronic invoicing system in Spain for business-to-business transactions, regulated by Royal Decree 1007/2023}
}

\newglossaryentry{lmtp}{
  name=LMTP,
  description={Local Mail Transfer Protocol, a protocol similar to SMTP used to deliver email from an MTA to a local mail server or mail processing system like Mailman Core}
}

\newglossaryentry{smtp}{
  name=SMTP,
  description={Simple Mail Transfer Protocol, the standard protocol used to send email across networks}
}

\newglossaryentry{restapi}{
  name=REST API,
  description={Application Programming Interface that follows the principles of Representational State Transfer (REST), enabling interaction with web services using HTTP methods}
}

\newglossaryentry{cicd}{
  name=CI/CD,
  description={Continuous Integration and Continuous Deployment, a development practice that automates testing, integration, and delivery of code changes}
}

\newglossaryentry{fxo}{
  name=FXO,
  description={Foreign Exchange Office. An interface that connects to the Public Switched Telephone Network (PSTN), such as a standard telephone wall socket, allowing analog lines to be used with VoIP systems}
}

\newglossaryentry{fxs}{
  name=FXS,
  description={Foreign Exchange Station. An interface that delivers analog line signals to devices like traditional phones or fax machines from a VoIP system}
}

\newglossaryentry{sip}{
  name=SIP,
  description={Session Initiation Protocol. A signaling protocol widely used for initiating, maintaining, and terminating real-time VoIP sessions}
}

\newglossaryentry{pbx}{
  name=PBX,
  description={Private Branch Exchange. A private telephone network used within an organization, enabling internal communication and access to external lines}
}

\newglossaryentry{glug}{
  name=GLUG,
  description={GNU/Linux User Group. A Linux User Group (LUG) or GNU/Linux User Group (GLUG) is a local, typically non-profit organization dedicated to supporting and educating Linux users, especially newcomers. In addition to promoting the use of the GNU/Linux operating system, these groups often advocate for free software, and in many cases also support free hardware and free culture as a whole}
}

\newglossaryentry{bbs}{
  name=BBS,
  description={A bulletin board system (BBS) is a computer server that users connect to using a terminal program. Once logged in, they can share files, read bulletins, and exchange messages via public forums or private messages. Before the internet became widespread, BBSs served as a vital platform for online interaction. Networks like FidoNet connected multiple BBSs, allowing users to send private messages (via a service called NetMail) across different systems, much like modern email}
}
